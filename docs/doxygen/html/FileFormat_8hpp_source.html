88inline FileHeader deserializeHeader(const std::vector<std::byte>& buffer) {
89 if (buffer.size() < HEADER_SIZE) {

93 size_t offset = 0;

96 // 1. Verify Magic Number
97 if (!std::equal(MAGIC_NUMBER.begin(), MAGIC_NUMBER.end(), buffer.begin() + offset)) {

99 }
100 offset += MAGIC_NUMBER.size();

102 // 2. Read and Verify Format Version
103 header.formatVersion = static_cast<uint8_t>(buffer[offset++]);
